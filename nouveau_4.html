// Effet de texte qui s’écrit lettre par lettre const slogan = "Infusé au rythme du vivant, à chaque gorgée."; const target = document.getElementById("slogan"); let i = 0; function typeWriter() { if (i < slogan.length) { target.textContent += slogan.charAt(i); i++; setTimeout(typeWriter, 80); } } document.addEventListener("DOMContentLoaded", typeWriter);
